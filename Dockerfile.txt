# =============================================
# Text Diff Checker - Docker Configuration
# =============================================

# استخدام Nginx كخادم للملفات الثابتة
FROM nginx:alpine

# معلومات الصورة
LABEL maintainer="developer@example.com"
LABEL version="1.0.0"
LABEL description="Text Diff Checker - أداة مقارنة النصوص"

# إنشاء مجلد العمل
WORKDIR /usr/share/nginx/html

# حذف الملفات الافتراضية
RUN rm -rf ./*

# نسخ ملفات المشروع
COPY index.html .
COPY assets/ ./assets/ 2>/dev/null || true

# نسخ إعدادات Nginx المخصصة (اختياري)
# COPY nginx.conf /etc/nginx/nginx.conf

# إعداد الصلاحيات
RUN chmod -R 755 /usr/share/nginx/html

# فتح المنفذ 80
EXPOSE 80

# تشغيل Nginx
CMD ["nginx", "-g", "daemon off;"]

# =============================================
# كيفية الاستخدام:
# 
# بناء الصورة:
#   docker build -t text-diff-checker .
#
# تشغيل الحاوية:
#   docker run -d -p 8080:80 --name diff-checker text-diff-checker
#
# الوصول للتطبيق:
#   http://localhost:8080
#
# إيقاف الحاوية:
#   docker stop diff-checker
#
# حذف الحاوية:
#   docker rm diff-checker
# =============================================
